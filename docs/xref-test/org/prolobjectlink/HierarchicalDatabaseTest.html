456 	public void testCreateQueryString() {
457 
458 		hdb.getTransaction().begin();
459 		hdb.delete(Point.class);
460 		hdb.delete(Segment.class);
461 		hdb.delete(Polygon.class);
462 		hdb.getTransaction().commit();
463 
464 		assertFalse(hdb.contains(a));
465 		assertFalse(hdb.contains(b));
466 		assertFalse(hdb.contains(c));
467 		assertFalse(hdb.contains(d));
468 
469 		hdb.insert(a, b, c, d);
470 		hdb.insert(ab, bc, ca, cd, da);
471 		hdb.insert(triangle, tetragon);
472 		hdb.getTransaction().commit();
473 
474 		assertTrue(hdb.contains(a));
475 		assertTrue(hdb.contains(b));
476 		assertTrue(hdb.contains(c));
477 		assertTrue(hdb.contains(d));
478 
479 		List<Object> objects = new ArrayList<Object>();
480 
481 		objects = hdb.createQuery("'" + Point.class.getName() + "'(Idp, X, Y)").getSolutions();
482 		assertEquals(Arrays.asList(a, b, c, d), createList(objects));
483 
484 		// point query restricted to specifics x and y points
485 		objects = hdb.createQuery("'" + Point.class.getName() + "'(Idp, X, Y), X =:= 3, Y =:= 14").getSolutions();
486 		assertEquals(Arrays.asList(a, b, c, d), createList(objects));
487 
488 		objects = hdb.createQuery("'" + Segment.class.getName() + "'(Ids, Point0, Point1)").getSolutions();
489 		assertEquals(Arrays.asList(ab, bc, ca, cd, da), createList(objects));
490 
491 		// segment query restricted to specifics points initial and finals
492 		objects = hdb
493 				.createQuery("'" + Segment.class.getName() + "'(Ids, Point0, Point1), Point0 == '"
494 						+ Point.class.getName() + "'(a, 3,14), Point1 == '" + Point.class.getName() + "'(b, 3,14)")
495 				.getSolutions();
496 		assertEquals(Arrays.asList(ab), createList(objects));
497 
498 		// predicate projection
499 		List<Point> points = new ArrayList<Point>();
500 		List<Segment> segments = new ArrayList<Segment>();
501 		objects = hdb.createQuery(
502 				"'" + Segment.class.getName() + "'(Ids, Point0, Point1), '" + Point.class.getName() + "'(Idp, X, Y)")
503 				.getSolutions();
504 		for (Object object : objects) {
505 			if (object instanceof Object[]) {
506 				Object[] array = (Object[]) object;
507 				if (array.length > 1) {
508 					Object object0 = array[0];
509 					Object object1 = array[1];
510 					if (object0 instanceof Segment) {
511 						Segment segment = (Segment) object0;
512 						segments.add(segment);
513 					}
514 					if (object1 instanceof Point) {
515 						Point point = (Point) object1;
516 						points.add(point);
517 					}
518 				}
519 			}
520 		}
521 
522 		// all combination created by the Projection or Cartesian Product
523 		assertEquals(Arrays.asList(ab, ab, ab, ab, bc, bc, bc, bc, ca, ca, ca, ca, cd, cd, cd, cd, da, da, da, da),
524 				segments);
525 		assertEquals(Arrays.asList(a, b, c, d, a, b, c, d, a, b, c, d, a, b, c, d, a, b, c, d), points);
526 
527 		hdb.getTransaction().close();
528 
529 	}

534 	public final void testCreateQueryString() {
535 
536 		storageManager.getTransaction().begin();
537 
538 		storageManager.delete(Point.class);
539 		storageManager.delete(Segment.class);
540 		storageManager.delete(Polygon.class);
541 
542 		assertFalse(storageManager.contains(a));
543 		assertFalse(storageManager.contains(b));
544 		assertFalse(storageManager.contains(c));
545 		assertFalse(storageManager.contains(d));
546 
547 		storageManager.insert(a, b, c, d);
548 		storageManager.insert(ab, bc, ca, cd, da);
549 		storageManager.insert(triangle, tetragon);
550 
551 		assertTrue(storageManager.contains(a));
552 		assertTrue(storageManager.contains(b));
553 		assertTrue(storageManager.contains(c));
554 		assertTrue(storageManager.contains(d));
555 
556 		List<Object> objects = new ArrayList<Object>();
557 
558 		objects = storageManager.createQuery("'" + Point.class.getName() + "'(Idp, X, Y)").getSolutions();
559 		assertEquals(Arrays.asList(a, b, c, d), createList(objects));
560 
561 		// point query restricted to specifics x and y points
562 		objects = storageManager.createQuery("'" + Point.class.getName() + "'(Idp, X, Y), X =:= 3, Y =:= 14")
563 				.getSolutions();
564 		assertEquals(Arrays.asList(a, b, c, d), createList(objects));
565 
566 		objects = storageManager.createQuery("'" + Segment.class.getName() + "'(Ids, Point0, Point1)").getSolutions();
567 		assertEquals(Arrays.asList(ab, bc, ca, cd, da), createList(objects));
568 
569 		// segment query restricted to specifics points initial and finals
570 		objects = storageManager
571 				.createQuery("'" + Segment.class.getName() + "'(Ids, Point0, Point1), Point0 == '"
572 						+ Point.class.getName() + "'(a, 3,14), Point1 == '" + Point.class.getName() + "'(b, 3,14)")
573 				.getSolutions();
574 		assertEquals(Arrays.asList(ab), createList(objects));
575 
576 		// predicate projection
577 		List<Point> points = new ArrayList<Point>();
578 		List<Segment> segments = new ArrayList<Segment>();
579 		objects = storageManager.createQuery(
580 				"'" + Segment.class.getName() + "'(Ids, Point0, Point1), '" + Point.class.getName() + "'(Idp, X, Y)")
581 				.getSolutions();
582 		for (Object object : objects) {
583 			if (object instanceof Object[]) {
584 				Object[] array = (Object[]) object;
585 				if (array.length > 1) {
586 					Object object0 = array[0];
587 					Object object1 = array[1];
588 					if (object0 instanceof Segment) {
589 						Segment segment = (Segment) object0;
590 						segments.add(segment);
591 					}
592 					if (object1 instanceof Point) {
593 						Point point = (Point) object1;
594 						points.add(point);
595 					}
596 				}
597 			}
598 		}
599 
600 		// all combination created by the Projection or Cartesian Product
601 		assertEquals(Arrays.asList(ab, ab, ab, ab, bc, bc, bc, bc, ca, ca, ca, ca, cd, cd, cd, cd, da, da, da, da),
602 				segments);
603 		assertEquals(Arrays.asList(a, b, c, d, a, b, c, d, a, b, c, d, a, b, c, d, a, b, c, d), points);
604 
605 		storageManager.getTransaction().commit();
606 		storageManager.getTransaction().close();
607 
608 	}

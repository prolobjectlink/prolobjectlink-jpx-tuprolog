319 	protected <S> List<S> createList(List<Object> solutions) {
320 		List<S> list = new ArrayList<S>();
321 		for (Object solution : solutions) {
322 			if (solution instanceof Object[]) {
323 				Object[] objects = (Object[]) solution;
324 				if (objects.length > 0) {
325 					Object object = objects[0];
326 					S s = (S) object;
327 					list.add(s);
328 				}
329 			}
330 		}
331 		return list;
332 	}
